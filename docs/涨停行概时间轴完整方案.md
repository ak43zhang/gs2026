# 涨停行概时间轴完整方案

## 问题总结

1. **概念表不存在** - `stock_concept`表不存在，应使用`cache_stock_industry_concept_bond`宽表
2. **数据查询方式** - 应沿用默认模式的宽表查询方式
3. **展示效果** - 应与默认模式完全一致（分组展示、命中标签等）

## 默认模式数据结构

### 宽表结构 (`cache_stock_industry_concept_bond`)
```sql
stock_code      VARCHAR(20)     -- 股票代码
stock_name      VARCHAR(100)    -- 股票名称
industry_codes  JSON            -- 行业代码列表
industry_names  JSON            -- 行业名称列表
concept_codes   JSON            -- 概念代码列表
concept_names   JSON            -- 概念名称列表
bond_code       VARCHAR(20)     -- 转债代码
bond_name       VARCHAR(100)    -- 转债名称
update_time     DATETIME        -- 更新时间
```

### 内存缓存结构 (`_stock_cache`)
```python
{
    'stock_code': {
        'stock_name': '股票名称',
        'industries': {'行业1', '行业2', ...},  # Set
        'concepts': {'概念1', '概念2', ...},     # Set
        'bond_code': '转债代码',
        'bond_name': '转债名称'
    }

```

### 默认模式查询流程

1. **获取涨停股票代码**
   - 当天：从`monitor_gp_sssj_{date}`查`is_zt=1`
   - 历史：从`analysis_ztb_detail_{year}`查`trade_date`

2. **从宽表缓存获取行业和概念**
   - 使用`_stock_cache`内存缓存
   - 遍历涨停股票，统计行业/概念频次

3. **筛选时查询实时价格**
   - 从`monitor_gp_sssj_{date}`获取最新时间点的价格

## 时间轴模式修改方案

### 核心修改

时间轴模式应该：
1. **查询指定时间点的涨停股票**（`ever_zt=1`且`time<=指定时间`）
2. **从宽表缓存获取行业和概念**（复用默认模式逻辑）
3. **查询指定时间点的实时价格**（不是最新时间）
4. **展示效果与默认模式完全一致**

```python
def _get_zt_stocks_at_time(date_str: str, time: str) -> List[str]:
    """获取到指定时间点为止的所有涨停股票（ever_zt=1）"""
    # 查询 monitor_gp_sssj 表中 ever_zt=1 且 time<=指定时间 的股票
    # 取每个股票在该时间点的最新记录
    sql = f"""
        SELECT stock_code, MAX(time) as max_time
        FROM monitor_gp_sssj_{date_str}
        WHERE ever_zt = 1 AND time <= '{time}'
        GROUP BY stock_code
    """
    # 或者使用子查询获取完整记录
    sql = f"""
        SELECT t1.*
        FROM monitor_gp_sssj_{date_str} t1
        INNER JOIN (
            SELECT stock_code, MAX(time) as max_time
            FROM monitor_gp_sssj_{date_str}
            WHERE ever_zt = 1 AND time <= '{time}'
            GROUP BY stock_code
        ) t2 ON t1.stock_code = t2.stock_code AND t1.time = t2.max_time
    """
```

```python
def _get_stock_details_batch(stock_codes: List[str]) -> Dict[str, Dict]:
    """批量获取股票详情（从宽表缓存，不是stock_concept）"""
    from gs2026.dashboard2.services.stock_picker_service import _stock_cache, load_memory_cache
    
    # 确保缓存已加载
    if not _stock_cache:
        load_memory_cache()
    
    result = {}
    for code in stock_codes:
        data = _stock_cache.get(code)
        if data:
            result[code] = {
                'industry_name': list(data['industries'])[0] if data['industries'] else '-',
                'industry_names': list(data['industries']),
                'concepts': list(data['concepts']),
                'bond_code': data.get('bond_code', '-'),
                'bond_name': data.get('bond_name', '-'),
            }
    
    return result
```

```python
def filter_ztb_snapshot(date: str, time: str, selected_tags: List[Dict], filters: Dict) -> Dict:
    """根据标签筛选涨停股票（时间轴版本）"""
    date_str = date.replace('-', '')
    
    # 1. 获取到指定时间点为止的所有涨停股票（含实时数据）
    df = _get_zt_stocks_data_at_time(date_str, time)  # 返回DataFrame含price, change_pct等
    
    if df.empty:
        return {'groups': [], 'summary': {'total': 0, 'with_bond': 0, 'query_time_ms': 0}}
    
    # 2. 从宽表缓存获取行业和概念
    stock_codes = df['stock_code'].astype(str).tolist()
    stock_details = _get_stock_details_batch(stock_codes)
    
    # 3. 根据标签筛选（复用默认模式逻辑）
    selected_industries = set(t['name'] for t in selected_tags if t['type'] == 'industry')
    selected_concepts = set(t['name'] for t in selected_tags if t['type'] == 'concept')
    
    stock_matches = {}
    for code in stock_codes:
        details = stock_details.get(code, {})
        all_tags = set(details.get('industry_names', [])) | set(details.get('concepts', []))
        matched = selected_industries | selected_concepts
        
        if matched & all_tags:  # 有交集
            matched_industries = list(selected_industries & set(details.get('industry_names', [])))
            matched_concepts = list(selected_concepts & set(details.get('concepts', [])))
            
            stock_matches[code] = {
                'stock_name': details.get('stock_name', ''),
                'bond_code': details.get('bond_code', '-'),
                'bond_name': details.get('bond_name', '-'),
                'industries': matched_industries,
                'concepts': matched_concepts,
                'match_count': len(matched_industries) + len(matched_concepts)
            }
    
    # 4. 查询转债价格（指定时间点）
    bond_codes = [m['bond_code'] for m in stock_matches.values() if m['bond_code'] != '-']
    bond_prices = _get_bond_prices_at_time(date_str, time, bond_codes)
    
    # 5. 组装结果并分组（与默认模式一致）
    groups = defaultdict(list)
    with_bond_count = 0
    
    for code, match_info in stock_matches.items():
        row = df[df['stock_code'] == code].iloc[0]
        
        stock_result = {
            'stock_code': code,
            'stock_name': match_info['stock_name'] or row.get('short_name', ''),
            'change_pct': float(row.get('change_pct', 0)),
            'price': float(row.get('price', 0)),
            'bond_code': match_info['bond_code'],
            'bond_name': match_info['bond_name'],
            'bond_change_pct': bond_prices.get(match_info['bond_code'], {}).get('change_pct', 0),
            'industry_name': '、'.join(match_info.get('all_industries', [])),
            'matched_industries': match_info['industries'],
            'matched_concepts': match_info['concepts'],
            'matched_tags_display': '\n'.join(match_info['industries'] + match_info['concepts'])
        }
        
        groups[match_info['match_count']].append(stock_result)
        
        if match_info['bond_code'] != '-':
            with_bond_count += 1
    
    # 6. 构建返回结果（与默认模式一致）
    result_groups = []
    for count in sorted(groups.keys(), reverse=True):
        stocks = groups[count]
        stocks.sort(key=lambda x: x['change_pct'], reverse=True)
        
        if count == len(selected_tags):
            label = f"命中全部 {count} 个"
        else:
            label = f"命中 {count} 个"
        
        result_groups.append({
            'match_count': count,
            'label': label,
            'stocks': stocks
        })
    
    return {
        'groups': result_groups,
        'summary': {
            'total': len(stock_matches),
            'with_bond': with_bond_count,
            'query_time_ms': 0
        }
    }
```

```python
def _get_zt_stocks_data_at_time(date_str: str, time: str) -> pd.DataFrame:
    """获取指定时间点涨停股票的完整数据（含price, change_pct等）"""
    # 先尝试Redis
    try:
        from gs2026.utils.redis_util import _get_redis_client
        redis_client = _get_redis_client()
        
        if redis_client:
            redis_key = f"monitor_gp_sssj_{date_str}:{time}"
            data = redis_client.get(redis_key)
            if data:
                import json
                df = pd.DataFrame(json.loads(data))
                if not df.empty:
                    # 筛选ever_zt=1的股票
                    df = df[df['ever_zt'] == 1]
                    return df
    except Exception as e:
        logger.warning(f"Redis查询失败: {e}")
    
    # 从MySQL获取
    try:
        table_name = f"monitor_gp_sssj_{date_str}"
        sql = f"""
            SELECT t1.*
            FROM {table_name} t1
            INNER JOIN (
                SELECT stock_code, MAX(time) as max_time
                FROM {table_name}
                WHERE ever_zt = 1 AND time <= '{time}'
                GROUP BY stock_code
            ) t2 ON t1.stock_code = t2.stock_code AND t1.time = t2.max_time
        """
        df = pd.read_sql(sql, engine)
        return df
    except Exception as e:
        logger.error(f"MySQL查询失败: {e}")
        return pd.DataFrame()


def _get_bond_prices_at_time(date_str: str, time: str, bond_codes: List[str]) -> Dict[str, Dict]:
    """获取指定时间点的转债价格"""
    if not bond_codes:
        return {}
    
    # 先尝试Redis
    try:
        from gs2026.utils.redis_util import _get_redis_client
        redis_client = _get_redis_client()
        
        if redis_client:
            redis_key = f"monitor_zq_sssj_{date_str}:{time}"
            data = redis_client.get(redis_key)
            if data:
                import json
                df = pd.DataFrame(json.loads(data))
                if not df.empty:
                    result = {}
                    for _, row in df.iterrows():
                        if row['bond_code'] in bond_codes:
                            result[row['bond_code']] = {
                                'price': row.get('price', 0),
                                'change_pct': float(row.get('change_pct', 0))
                            }
                    return result
    except Exception as e:
        logger.warning(f"Redis查询转债失败: {e}")
    
    # 从MySQL获取
    try:
        table_name = f"monitor_zq_sssj_{date_str}"
        placeholders = ','.join([f"'{code}'" for code in bond_codes])
        sql = f"""
            SELECT bond_code, price, change_pct
            FROM {table_name}
            WHERE time = '{time}' AND bond_code IN ({placeholders})
        """
        df = pd.read_sql(sql, engine)
        return {row['bond_code']: {'price': row['price'], 'change_pct': float(row['change_pct'])} 
                for _, row in df.iterrows()}
    except Exception as e:
        logger.error(f"MySQL查询转债失败: {e}")
        return {}
```
