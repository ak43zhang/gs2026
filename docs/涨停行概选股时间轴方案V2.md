# 涨停行概选股时间轴方案设计 V2

## 一、需求理解（更新）

### 用户新思路
1. **点击时间轴时间** → 获取该时间点
2. **查询数据源**: `monitor_gp_sssj_{date}`（先Redis，后MySQL）
3. **筛选条件**: `is_zt = 1`（该时间点所有涨停股票）
4. **后续分析**: 对这些涨停股票进行行概分析

### 关键变化
- **数据源变化**: 从`zt_records`表 → `monitor_gp_sssj`实时数据表
- **筛选逻辑变化**: 从`first_zt_time` → `is_zt=1`字段
- **数据时效性**: 使用实时计算的数据（每3秒更新）

### 方案本质
**"某时间点的市场涨停快照"** —— 查看该时刻市场上所有涨停的股票

---

## 二、方案设计 V2

### 2.1 数据流

```
用户点击时间轴时间(如10:00)
    ↓
前端传递: date=2026-04-29, time=10:00:00
    ↓
后端查询:
    1. 先查Redis: key="monitor_gp_sssj_20260429:10:00:00"
    2. Redis未命中: 查MySQL表 monitor_gp_sssj_20260429
    3. 筛选: is_zt = 1
    ↓
获取该时间点所有涨停股票列表
    ↓
行概分析:
    - 行业分布
    - 概念分布
    - 联动关系
    - ...
    ↓
返回结果给前端展示
```

### 2.2 核心优势

| 优势 | 说明 |
|------|------|
| **实时性** | 使用实时计算的数据（3秒更新），非历史记录 |
| **准确性** | `is_zt`字段实时计算，反映真实市场状态 |
| **一致性** | 与数据监控页面使用同一数据源 |
| **灵活性** | 可查询任意时间点的涨停状态 |

### 3.3 服务层实现

```python
# ztb_analysis_service.py

def get_ztb_list_by_time(date: str, time_str: str) -> Dict:
    """
    获取指定时间点的涨停股票列表
    
    查询流程:
    1. 先查Redis
    2. Redis未命中，查MySQL
    3. 筛选is_zt=1的股票
    4. 进行行概分析
    
    Args:
        date: 日期，如 "2026-04-29"
        time_str: 时间，如 "10:00:00"
    
    Returns:
        {
            'zt_stocks': [...],        # 涨停股票列表
            'industry_analysis': {...}, # 行业分析
            'concept_analysis': {...},  # 概念分析
            'stats': {...}              # 统计信息
        }
    """
    # 1. 构建Redis key
    date_str = date.replace('-', '')
    redis_key = f"monitor_gp_sssj_{date_str}:{time_str}"
    
    df = None
    source = None
    
    # 2. 先查Redis
    try:
        df = redis_util.load_dataframe_by_key(redis_key)
        if df is not None and not df.empty:
            source = 'redis'
            logger.info(f"从Redis获取数据: {redis_key}, {len(df)}条")
    except Exception as e:
        logger.warning(f"Redis查询失败: {e}")
    
    # 3. Redis未命中，查MySQL
    if df is None or df.empty:
        try:
            table_name = f"monitor_gp_sssj_{date_str}"
            sql = f"""
                SELECT * FROM {table_name}
                WHERE time = %s
            """
            df = pd.read_sql(sql, engine, params=(time_str,))
            source = 'mysql'
            logger.info(f"从MySQL获取数据: {table_name}, time={time_str}, {len(df)}条")
        except Exception as e:
            logger.error(f"MySQL查询失败: {e}")
            raise
    
    if df is None or df.empty:
        return {
            'zt_stocks': [],
            'industry_analysis': {},
            'concept_analysis': {},
            'stats': {
                'total': 0,
                'source': None,
                'time': time_str
            }
        }
    
    # 4. 筛选is_zt=1的股票
    if 'is_zt' not in df.columns:
        logger.error(f"数据缺少is_zt字段，可用字段: {df.columns.tolist()}")
        raise ValueError("数据缺少is_zt字段")
    
    zt_df = df[df['is_zt'] == 1].copy()
    
    logger.info(f"时间点{time_str}涨停股票: {len(zt_df)}/{len(df)}")
    
    # 5. 基础字段处理
    zt_stocks = []
    for _, row in zt_df.iterrows():
        zt_stocks.append({
            'stock_code': str(row.get('stock_code', '')).zfill(6),
            'stock_name': row.get('short_name', ''),
            'price': float(row.get('price', 0)),
            'change_pct': float(row.get('change_pct', 0)),
            'main_net_amount': float(row.get('main_net_amount', 0)),
            'cumulative_main_net': float(row.get('cumulative_main_net', 0)),
            'time': time_str
        })
    
    # 6. 行概分析
    industry_analysis = analyze_industry_distribution(zt_stocks)
    concept_analysis = analyze_concept_distribution(zt_stocks)
    
    return {
        'zt_stocks': zt_stocks,
        'industry_analysis': industry_analysis,
        'concept_analysis': concept_analysis,
        'stats': {
            'total': len(zt_stocks),
            'source': source,
            'time': time_str,
            'date': date
        }
    }


def analyze_industry_distribution(stocks: List[Dict]) -> Dict:
    """分析涨停股票的行业分布"""
    if not stocks:
        return {}
    
    # 获取股票行业信息（从缓存或数据库）
    stock_codes = [s['stock_code'] for s in stocks]
    industry_map = get_stock_industries(stock_codes)
    
    # 统计行业分布
    industry_count = {}
    for stock in stocks:
        code = stock['stock_code']
        industry = industry_map.get(code, '未知')
        industry_count[industry] = industry_count.get(industry, 0) + 1
    
    # 排序
    sorted_industries = sorted(
        industry_count.items(), 
        key=lambda x: x[1], 
        reverse=True
    )
    
    return {
        'distribution': dict(sorted_industries),
        'top_industries': sorted_industries[:5]
    }


def analyze_concept_distribution(stocks: List[Dict]) -> Dict:
    """分析涨停股票的概念分布"""
    if not stocks:
        return {}
    
    # 获取股票概念信息
    stock_codes = [s['stock_code'] for s in stocks]
    concept_map = get_stock_concepts(stock_codes)
    
    # 统计概念分布
    concept_count = {}
    for stock in stocks:
        code = stock['stock_code']
        concepts = concept_map.get(code, [])
        for concept in concepts:
            concept_count[concept] = concept_count.get(concept, 0) + 1
    
    # 排序
    sorted_concepts = sorted(
        concept_count.items(),
        key=lambda x: x[1],
        reverse=True
    )
    
    return {
        'distribution': dict(sorted_concepts),
        'top_concepts': sorted_concepts[:10]
    }
```

// stock_picker.js 新增逻辑

// 时间轴配置
const TIMELINE_CONFIG = {
    startTime: '09:30',
    endTime: '15:00',
    interval: 3,  // 3分钟一个点
    totalPoints: 120  // (15:00-09:30) / 3min = 120个点
};

// 初始化时间轴
function initTimeline() {
    const slider = document.getElementById('timelineSlider');
    const timeDisplay = document.getElementById('timelineTime');
    
    // 滑块变化事件
    slider.addEventListener('input', function() {
        const point = parseInt(this.value);
        const timeStr = pointToTime(point);
        timeDisplay.textContent = timeStr;
    });
    
    // 刷新按钮
    document.getElementById('refreshBtn').addEventListener('click', function() {
        loadZtData();
    });
    
    // 快速跳转按钮
    document.querySelectorAll('.quick-jump button').forEach(btn => {
        btn.addEventListener('click', function() {
            const timeStr = this.dataset.time;
            const point = timeToPoint(timeStr);
            slider.value = point;
            slider.dispatchEvent(new Event('input'));
            loadZtData();
        });
    });
    
    // 实时模式
    let liveTimer = null;
    document.getElementById('liveBtn').addEventListener('click', function() {
        if (liveTimer) {
            clearInterval(liveTimer);
            liveTimer = null;
            this.textContent = '实时';
            this.classList.remove('active');
        } else {
            this.textContent = '暂停';
            this.classList.add('active');
            liveTimer = setInterval(() => {
                const slider = document.getElementById('timelineSlider');
                let point = parseInt(slider.value) + 1;
                if (point > TIMELINE_CONFIG.totalPoints) {
                    point = 0;
                }
                slider.value = point;
                slider.dispatchEvent(new Event('input'));
                loadZtData();
            }, 3000);  // 3秒刷新
        }
    });
}

// 时间点转字符串
function pointToTime(point) {
    const [startHour, startMin] = TIMELINE_CONFIG.startTime.split(':').map(Number);
    const totalMinutes = startHour * 60 + startMin + point * TIMELINE_CONFIG.interval;
    
    // 处理午休
    if (totalMinutes >= 11 * 60 + 30 && totalMinutes < 13 * 60) {
        // 午休时间，跳到13:00
        return '13:00';
    }
    
    const hour = Math.floor(totalMinutes / 60);
    const min = totalMinutes % 60;
    return `${hour.toString().padStart(2, '0')}:${min.toString().padStart(2, '0')}`;
}

// 时间字符串转点
function timeToPoint(timeStr) {
    const [hour, min] = timeStr.split(':').map(Number);
    const [startHour, startMin] = TIMELINE_CONFIG.startTime.split(':').map(Number);
    
    let totalMinutes = hour * 60 + min;
    const startTotal = startHour * 60 + startMin;
    
    // 处理午休
    if (hour === 13 && min < 30) {
        totalMinutes = 11 * 60 + 30;  // 跳到午休前
    }
    
    return Math.floor((totalMinutes - startTotal) / TIMELINE_CONFIG.interval);
}

// 加载涨停数据
async function loadZtData() {
    const date = document.getElementById('datePicker').value;
    const time = document.getElementById('timelineTime').textContent;
    
    try {
        const response = await fetch(`/api/ztb/list-by-time?date=${date}&time=${time}:00`);
        const result = await response.json();
        
        if (result.code === 0) {
            renderZtList(result.data.zt_stocks);
            renderIndustryAnalysis(result.data.industry_analysis);
            renderConceptAnalysis(result.data.concept_analysis);
            updateStats(result.data.stats);
        } else {
            showError(result.message);
        }
    } catch (error) {
        showError('加载数据失败: ' + error.message);
    }
}

// 渲染涨停列表
function renderZtList(stocks) {
    const tbody = document.querySelector('#ztTable tbody');
    tbody.innerHTML = '';
    
    stocks.forEach(stock => {
        const row = document.createElement('tr');
        row.innerHTML = `
            <td>${stock.stock_code}</td>
            <td>${stock.stock_name}</td>
            <td>${stock.price.toFixed(2)}</td>
            <td class="${stock.change_pct > 0 ? 'up' : 'down'}">
                ${stock.change_pct > 0 ? '+' : ''}${stock.change_pct.toFixed(2)}%
            </td>
            <td>${formatMainNet(stock.main_net_amount)}</td>
            <td>${stock.industry || '-'}</td>
        `;
        tbody.appendChild(row);
    });
    
    document.getElementById('ztCount').textContent = stocks.length;
}

// 格式化主力净额
function formatMainNet(amount) {
    if (amount === 0) return '-';
    const wan = amount / 10000;
    return (amount > 0 ? '+' : '') + wan.toFixed(1) + '万';
}
