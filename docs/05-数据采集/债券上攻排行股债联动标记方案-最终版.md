## 1. 需求概述

### 1.1 背景
债券上攻排行中，某些可转债对应的正股可能同时出现在股票上攻排行中。需要对这种"股债联动"情况进行视觉标记，便于快速识别强势联动标的。

### 1.2 标记规则
**触发条件**：债券的正股代码出现在当前（或最近）股票上攻排行中。

**标记方式**：
- 前缀标记：`★[排名]债券代码`
- 示例：`★[3]128123` 表示正股在股票排行第3位

---

## 2. 核心挑战

### 2.1 挑战一：时序问题
股票和债券监控可能异步执行，导致债券计算时股票排行数据尚未生成。

**解决方案**：延迟标记法（带降级）

### 2.2 挑战二：正股代码缺失【关键】
**问题**：债券数据源（adata）**不直接提供正股代码字段**。

**解决方案**：通过外部数据源获取债券→正股映射

### 3.2 实现方案

```python
# ========== 新增：债券正股映射获取 ==========

_bond_stock_mapping = None
_bond_mapping_date = None

def get_bond_stock_mapping() -> dict:
    """
    获取债券代码到正股代码的映射
    
    Returns:
        Dict[债券代码, 正股代码]
    """
    try:
        import akshare as ak
        
        # 方法1：使用AKShare获取可转债列表
        df = ak.bond_cb_jsl()
        if 'bond_id' in df.columns and 'stock_id' in df.columns:
            mapping = dict(zip(
                df['bond_id'].astype(str).str.zfill(6),
                df['stock_id'].astype(str).str.zfill(6)
            ))
            logger.info(f"[股债联动] AKShare获取债券映射: {len(mapping)}条")
            return mapping
    except Exception as e:
        logger.warning(f"[股债联动] AKShare获取债券映射失败: {e}")
    
    try:
        # 方法2：使用AData
        import adata
        df = adata.bond.info.all_bonds()
        if 'bond_code' in df.columns and 'stock_code' in df.columns:
            mapping = dict(zip(
                df['bond_code'].astype(str).str.zfill(6),
                df['stock_code'].astype(str).str.zfill(6)
            ))
            logger.info(f"[股债联动] AData获取债券映射: {len(mapping)}条")
            return mapping
    except Exception as e:
        logger.warning(f"[股债联动] AData获取债券映射失败: {e}")
    
    return {}


def get_cached_bond_stock_mapping(force_refresh=False) -> dict:
    """获取缓存的债券正股映射（每天自动刷新）"""
    global _bond_stock_mapping, _bond_mapping_date
    
    from datetime import datetime
    today = datetime.now().strftime('%Y%m%d')
    
    if force_refresh or _bond_stock_mapping is None or _bond_mapping_date != today:
        _bond_stock_mapping = get_bond_stock_mapping()
        _bond_mapping_date = today
        logger.info(f"[股债联动] 加载债券正股映射: {len(_bond_stock_mapping)}条")
    
    return _bond_stock_mapping
```

#### 函数1：健壮获取股票排行代码

```python
def get_stock_attack_codes_robust(
    date_str: str, 
    time_str: str,
    max_lookback: int = None
) -> Dict[str, int]:
    """
    健壮地获取股票上攻排行代码（带降级处理）
    
    依次尝试：
    1. 当前周期实时数据
    2. 上一周期数据（3秒前）
    3. 最近N个周期内的数据
    4. 返回空（放弃标记）
    """
    from datetime import datetime, timedelta
    
    if max_lookback is None:
        max_lookback = STOCK_BOND_LINK_CONFIG['max_lookback_periods']
    
    try:
        current_dt = datetime.strptime(f"{date_str} {time_str}", "%Y%m%d %H:%M:%S")
    except ValueError:
        logger.error(f"[股债联动] 时间格式错误: {date_str} {time_str}")
        return {}
    
    # 依次尝试当前周期和回看周期
    for i in range(max_lookback + 1):
        try_time = current_dt - timedelta(seconds=3 * i)
        try_time_str = try_time.strftime("%H:%M:%S")
        
        codes = _get_stock_codes_from_redis(date_str, try_time_str)
        if codes:
            if i == 0:
                logger.debug(f"[股债联动] 使用实时股票排行: {try_time_str}, {len(codes)}只")
            else:
                logger.info(f"[股债联动] 使用{i}周期前股票排行: {try_time_str}, {len(codes)}只")
            return codes
    
    # 最后尝试：获取任何可用的股票排行
    codes = _get_latest_stock_codes(date_str)
    if codes:
        logger.info(f"[股债联动] 使用最新可用股票排行（非实时）, {len(codes)}只")
        return codes
    
    logger.warning(f"[股债联动] 无法获取股票排行，跳过标记")
    return {}


def _get_stock_codes_from_redis(date_str: str, time_str: str) -> Dict[str, int]:
    """从Redis获取股票排行代码"""
    codes = {}
    
    try:
        # 方法1：从rank key获取
        rank_key = f"rank:stock:code_{date_str}"
        r = redis_util._get_redis_client()
        rank_data = r.zrevrange(rank_key, 0, -1, withscores=True)
        
        for idx, (code, score) in enumerate(rank_data, 1):
            code_str = code.decode('utf-8') if isinstance(code, bytes) else code
            codes[code_str] = idx
        
        if codes:
            return codes
            
        # 方法2：从top30表获取
        gp_table = f"monitor_gp_top30_{date_str}"
        df = redis_util.load_dataframe_by_key(
            f"{gp_table}:{time_str}", 
            use_compression=False
        )
        if df is not None and not df.empty:
            for idx, row in df.iterrows():
                code = str(row.get('code', ''))
                if code:
                    codes[code] = idx + 1
                    
    except Exception as e:
        logger.debug(f"[股债联动] 从Redis获取股票排行失败: {e}")
    
    return codes


def _get_latest_stock_codes(date_str: str) -> Dict[str, int]:
    """获取最近可用的股票排行"""
    codes = {}
    
    try:
        gp_table = f"monitor_gp_top30_{date_str}"
        pattern = f"{gp_table}:*"
        r = redis_util._get_redis_client()
        keys = r.keys(pattern)
        
        if not keys:
            return {}
        
        # 找到最新的时间戳
        latest_time = None
        for key in keys:
            key_str = key.decode('utf-8') if isinstance(key, bytes) else key
            time_part = key_str.split(':', 1)[1] if ':' in key_str else None
            if time_part and (latest_time is None or time_part > latest_time):
                latest_time = time_part
        
        if latest_time:
            return _get_stock_codes_from_redis(date_str, latest_time)
            
    except Exception as e:
        logger.warning(f"[股债联动] 获取最近股票排行失败: {e}")
    
    return codes
```

#### 函数2：标记债券代码

```python
def mark_bond_with_stock_link(df_bond: pd.DataFrame, stock_codes: dict) -> pd.DataFrame:
    """
    为债券添加股债联动标记
    
    Args:
        df_bond: 债券数据DataFrame（需包含stock_code字段）
        stock_codes: 股票排行代码字典 {股票代码: 排名}
    
    Returns:
        添加标记后的DataFrame
    """
    if not stock_codes:
        # 无股票排行数据，添加空标记
        df_bond['is_stock_ranked'] = False
        df_bond['stock_rank_pos'] = None
        df_bond['bond_code_display'] = df_bond['code'].astype(str)
        return df_bond
    
    # 检查正股是否在股票排行中
    df_bond['stock_rank_pos'] = df_bond['stock_code'].map(stock_codes)
    df_bond['is_stock_ranked'] = df_bond['stock_rank_pos'].notna()
    
    # 生成标记后的债券代码
    def _mark_code(row):
        if row['is_stock_ranked']:
            prefix = STOCK_BOND_LINK_CONFIG['mark_prefix']
            if STOCK_BOND_LINK_CONFIG['include_rank'] and pd.notna(row['stock_rank_pos']):
                # 包含排名：★[3]128123
                return f"{prefix}[{int(row['stock_rank_pos'])}]{row['code']}"
            else:
                # 仅前缀：★128123
                return f"{prefix}{row['code']}"
        return str(row['code'])
    
    df_bond['bond_code_display'] = df_bond.apply(_mark_code, axis=1)
    
    # 记录统计
    ranked_count = df_bond['is_stock_ranked'].sum()
    if ranked_count > 0:
        logger.info(f"[股债联动] 标记{int(ranked_count)}只债券（正股在上攻排行中）")
    
    return df_bond
```
