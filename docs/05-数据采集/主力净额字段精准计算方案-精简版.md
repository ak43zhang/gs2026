## 2. 核心计算逻辑

### 2.1 基本假设

```python
# 假设1：价格上涨时，主动买入力量 > 主动卖出力量
# 假设2：价格下跌时，主动卖出力量 > 主动买入力量
# 假设3：成交量放大时，主力参与度更高
# 假设4：价格变化越大，主力主导程度越高

# 主力净额 = 成交额 × 主力参与系数 × 方向系数
```

### 2.2 主力参与系数计算

```python
def calculate_main_participation_ratio(delta_amount, delta_volume, 
                                        avg_amount_20d, avg_volume_20d):
    """
    计算主力参与系数（0-1之间）
    
    逻辑：
    - 成交额越大，主力参与概率越高
    - 成交量越大，主力参与概率越高
    - 超过历史均值越多，主力参与概率越高
    """
    
    # 绝对阈值门槛
    MIN_AMOUNT = 300000  # 30万
    MIN_VOLUME = 20000   # 200手
    
    # 门槛检查
    if delta_amount < MIN_AMOUNT or delta_volume < MIN_VOLUME:
        return 0.0  # 未达到主力门槛，认定无主力参与
    
    # 计算相对历史均值的比例
    amount_ratio = delta_amount / avg_amount_20d if avg_amount_20d > 0 else 1.0
    volume_ratio = delta_volume / avg_volume_20d if avg_volume_20d > 0 else 1.0
    
    # 基础参与系数（基于成交额）
    # 30万 -> 0.3, 50万 -> 0.5, 100万 -> 0.8, 200万 -> 1.0
    if delta_amount >= 2000000:  # 200万
        base_ratio = 1.0
    elif delta_amount >= 1000000:  # 100万
        base_ratio = 0.8 + (delta_amount - 1000000) / 1000000 * 0.2
    elif delta_amount >= 500000:  # 50万
        base_ratio = 0.5 + (delta_amount - 500000) / 500000 * 0.3
    else:  # 30-50万
        base_ratio = 0.3 + (delta_amount - 300000) / 200000 * 0.2
    
    # 量能放大系数
    # 如果成交量超过均值2倍，增加参与系数
    volume_boost = min(0.2, (volume_ratio - 1) * 0.1) if volume_ratio > 1 else 0
    
    # 综合参与系数
    participation_ratio = min(1.0, base_ratio + volume_boost)
    
    return participation_ratio
```

### 2.3 方向系数计算

```python
def calculate_direction_coefficient(price_change_pct, delta_amount, 
                                   avg_amount_20d, volatility_20d):
    """
    计算方向系数（-1到+1之间）
    
    逻辑：
    - 价格大幅上涨 -> 主力买入（正值）
    - 价格大幅下跌 -> 主力卖出（负值）
    - 价格平稳但放量 -> 根据量能判断（可能是吸筹或出货）
    """
    
    # 情况1：价格明显上涨（>0.1%）
    if price_change_pct > 0.1:
        # 涨幅越大，买入确定性越高
        if price_change_pct >= 1.0:  # 涨超1%
            direction = 1.0  # 确定是买入
        elif price_change_pct >= 0.5:  # 涨0.5%-1%
            direction = 0.8 + (price_change_pct - 0.5) / 0.5 * 0.2
        else:  # 涨0.1%-0.5%
            direction = 0.5 + (price_change_pct - 0.1) / 0.4 * 0.3
    
    # 情况2：价格明显下跌（<-0.1%）
    elif price_change_pct < -0.1:
        # 跌幅越大，卖出确定性越高
        if price_change_pct <= -1.0:  # 跌超1%
            direction = -1.0  # 确定是卖出
        elif price_change_pct <= -0.5:  # 跌0.5%-1%
            direction = -(0.8 + abs(price_change_pct + 0.5) / 0.5 * 0.2)
        else:  # 跌0.1%-0.5%
            direction = -(0.5 + abs(price_change_pct + 0.1) / 0.4 * 0.3)
    
    # 情况3：价格基本平稳（-0.1%到+0.1%）
    else:
        # 价格平稳但放量，可能是吸筹或出货
        # 根据历史位置判断（需要更多信息）
        # 简化处理：认定为中性，方向系数为0
        direction = 0.0
    
    return direction
```

### 2.4 主力净额计算

```python
def calculate_main_net_amount_v3(stock_code, current_data, previous_data, historical_data):
    """
    计算主力净额（改进版）
    
    公式：主力净额 = 周期成交额 × 主力参与系数 × 方向系数
    
    Args:
        stock_code: 股票代码
        current_data: 当前时刻数据
        previous_data: 上一时刻数据
        historical_data: 历史统计数据
    
    Returns:
        main_net_amount: 主力净额（正值=净流入，负值=净流出）
    """
    
    # 计算周期变化
    delta_amount = current_data['amount'] - previous_data['amount']
    delta_volume = current_data['volume'] - previous_data['volume']
    price_change_pct = current_data['change_pct'] - previous_data.get('change_pct', 0)
    
    # 获取历史统计
    avg_amount_20d = historical_data.get('avg_amount_20d', delta_amount)
    avg_volume_20d = historical_data.get('avg_volume_20d', delta_volume)
    volatility_20d = historical_data.get('volatility_20d', 0.01)
    
    # 计算主力参与系数
    participation_ratio = calculate_main_participation_ratio(
        delta_amount, delta_volume, avg_amount_20d, avg_volume_20d
    )
    
    # 如果参与系数为0，直接返回0
    if participation_ratio <= 0:
        return 0.0
    
    # 计算方向系数
    direction = calculate_direction_coefficient(
        price_change_pct, delta_amount, avg_amount_20d, volatility_20d
    )
    
    # 计算主力净额
    # 只有参与系数和方向系数都有效时才计算
    if abs(direction) > 0:
        main_net_amount = delta_amount * participation_ratio * direction
    else:
        # 方向不明确时，根据量能判断
        if delta_amount > avg_amount_20d * 2:  # 放量超过2倍
            # 放量但价格平稳，可能是主力在建仓或出货
            # 简化处理：认定为吸筹（小幅正值）
            main_net_amount = delta_amount * participation_ratio * 0.3
        else:
            main_net_amount = 0.0
    
    return round(main_net_amount, 2)
```

## 3. 关键改进点

### 3.1 与之前方案的对比

| 方面 | 原方案 | 改进方案 |
|------|--------|----------|
| 主力认定 | 固定阈值（30万） | 渐变系数（30万->0.3, 200万->1.0） |
| 方向判断 | 只看价格涨跌 | 根据涨跌幅度计算方向系数（0.5-1.0） |
| 金额计算 | 全额计入或0 | 按参与系数和方向系数加权 |
| 中性情况 | 未处理 | 放量但价格平稳时给予小幅度值 |

MAIN_FORCE_CONFIG = {
    # 门槛值
    'min_amount': 300000,      # 30万
    'min_volume': 20000,       # 200手
    
    # 参与系数参数
    'participation_thresholds': {

        'level3': {'amount': 1000000, 'ratio': 0.8},  # 100万 -> 0.8

    'direction_thresholds': {
        'strong_up': 1.0,     # 涨>1% -> 方向1.0
        'medium_up': 0.5,     # 涨0.5% -> 方向0.8
        'weak_up': 0.1,       # 涨0.1% -> 方向0.5
        'neutral': 0.0,       # 平稳 -> 方向0
        'weak_down': -0.1,    # 跌0.1% -> 方向-0.5
        'medium_down': -0.5,  # 跌0.5% -> 方向-0.8
        'strong_down': -1.0,  # 跌>1% -> 方向-1.0
    },
    
    # 量能放大系数
    'volume_boost_max': 0.2,   # 最大加0.2
    'volume_boost_ratio': 0.1,  # 每超1倍均值加0.1

### 5.2 计算函数

```python
def calculate_main_net_amount(stock_code, df_now, df_prev, historical_stats):
    """
    计算主力净额
    
    Args:
        stock_code: 股票代码
        df_now: 当前时刻DataFrame
        df_prev: 上一时刻DataFrame（可能为None）
        historical_stats: 历史统计数据
    
    Returns:
        float: 主力净额（元）
    """
    if df_prev is None or df_prev.empty:
        return 0.0
    
    # 合并数据
    merged = pd.merge(
        df_now[['stock_code', 'price', 'volume', 'amount', 'change_pct']],
        df_prev[['stock_code', 'price', 'volume', 'amount', 'change_pct']],
        on='stock_code',
        suffixes=('_now', '_prev'),
        how='inner'
    )
    
    if merged.empty:
        return 0.0
    
    # 获取当前股票数据
    stock_data = merged[merged['stock_code'] == stock_code]
    if stock_data.empty:
        return 0.0
    
    row = stock_data.iloc[0]
    
    # 计算周期变化
    delta_amount = row['amount_now'] - row['amount_prev']
    delta_volume = row['volume_now'] - row['volume_prev']
    price_change_pct = row['change_pct_now'] - row['change_pct_prev']
    
    # 获取历史统计
    avg_amount_20d = historical_stats.get('avg_amount_20d', delta_amount)
    avg_volume_20d = historical_stats.get('avg_volume_20d', delta_volume)
    
    # 门槛检查
    if delta_amount < MAIN_FORCE_CONFIG['min_amount']:
        return 0.0
    if delta_volume < MAIN_FORCE_CONFIG['min_volume']:
        return 0.0
    
    # 计算参与系数
    participation = calculate_participation_ratio(
        delta_amount, delta_volume, avg_amount_20d, avg_volume_20d
    )
    
    # 计算方向系数
    direction = calculate_direction_coefficient(
        price_change_pct, delta_amount, avg_amount_20d
    )
    
    # 计算主力净额
    if abs(direction) > 0:
        main_net = delta_amount * participation * direction
    else:
        # 方向不明确，但放量超过2倍
        if delta_amount > avg_amount_20d * 2:
            main_net = delta_amount * participation * 0.3
        else:
            main_net = 0.0
    
    return round(main_net, 2)


def calculate_participation_ratio(delta_amount, delta_volume, avg_amount, avg_volume):
    """计算主力参与系数"""
    thresholds = MAIN_FORCE_CONFIG['participation_thresholds']
    
    # 基础系数
    if delta_amount >= thresholds['level4']['amount']:
        base = thresholds['level4']['ratio']
    elif delta_amount >= thresholds['level3']['amount']:
        base = thresholds['level3']['ratio'] + \
               (delta_amount - thresholds['level3']['amount']) / \
               (thresholds['level4']['amount'] - thresholds['level3']['amount']) * \
               (thresholds['level4']['ratio'] - thresholds['level3']['ratio'])
    elif delta_amount >= thresholds['level2']['amount']:
        base = thresholds['level2']['ratio'] + \
               (delta_amount - thresholds['level2']['amount']) / \
               (thresholds['level3']['amount'] - thresholds['level2']['amount']) * \
               (thresholds['level3']['ratio'] - thresholds['level2']['ratio'])
    else:
        base = thresholds['level1']['ratio'] + \
               (delta_amount - thresholds['level1']['amount']) / \
               (thresholds['level2']['amount'] - thresholds['level1']['amount']) * \
               (thresholds['level2']['ratio'] - thresholds['level1']['ratio'])
    
    # 量能放大系数
    if avg_volume > 0:
        volume_ratio = delta_volume / avg_volume
        boost = min(MAIN_FORCE_CONFIG['volume_boost_max'], 
                    (volume_ratio - 1) * MAIN_FORCE_CONFIG['volume_boost_ratio'])
    else:
        boost = 0
    
    return min(1.0, base + boost)


def calculate_direction_coefficient(price_change_pct, delta_amount, avg_amount):
    """计算方向系数"""
    thresholds = MAIN_FORCE_CONFIG['direction_thresholds']
    
    if price_change_pct >= 1.0:
        return thresholds['strong_up']
    elif price_change_pct >= 0.5:
        return thresholds['medium_up'] + (price_change_pct - 0.5) / 0.5 * 0.2
    elif price_change_pct >= 0.1:
        return thresholds['weak_up'] + (price_change_pct - 0.1) / 0.4 * 0.3
    elif price_change_pct <= -1.0:
        return thresholds['strong_down']
    elif price_change_pct <= -0.5:
        return -(0.8 + abs(price_change_pct + 0.5) / 0.5 * 0.2)
    elif price_change_pct <= -0.1:
        return -(0.5 + abs(price_change_pct + 0.1) / 0.4 * 0.3)
    else:
        # 价格平稳
        if delta_amount > avg_amount * 2:
            # 放量但价格平稳，可能是吸筹
            return 0.3
        else:
            return 0.0
```

### 7.1 预期准确率

| 场景 | 准确率 | 说明 |
|------|--------|------|

## 8. 总结

### 8.1 方案特点

| 特点 | 说明 |
|------|------|
| 精简 | 只计算一个字段（main_net_amount） |
| 渐进 | 根据成交额大小渐进计算参与系数 |
| 方向敏感 | 根据价格变化幅度计算方向系数 |
| 量能加权 | 成交量放大时增加参与系数 |
| 中性处理 | 放量但价格平稳时给予保守估计 |

### 8.2 计算公式

```
主力净额 = 周期成交额 × 主力参与系数 × 方向系数

其中：
- 主力参与系数：0-1之间，根据成交额大小和量能放大程度计算
- 方向系数：-1到+1之间，根据价格变化幅度计算
```
