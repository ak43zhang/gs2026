## 1. 当前方案深度分析

### 1.1 3秒周期的问题

**核心问题**：3秒周期太短，无法准确识别主力行为

```
问题分析：
├── 3秒内可能包含多个交易方向
│   ├── 主力买入 + 散户卖出
│   ├── 主力卖出 + 散户买入
│   └── 多个主力账户对倒
│
├── 3秒累计数据的局限
│   ├── 无法识别单笔大单
│   ├── 无法区分主动/被动成交
│   └── 无法识别买卖方向
│
└── 主力行为特征（需要更长时间窗口）
    ├── 吸筹：持续小单买入，持续数分钟到数小时
    ├── 拉升：快速大单买入，持续30秒-5分钟
    ├── 出货：持续大单卖出，持续数分钟到数小时
    └── 对倒：自买自卖，需要识别关联账户
```

**3秒周期 vs 主力行为时间尺度**：

| 主力行为 | 典型持续时间 | 3秒周期能否识别 |
|----------|-------------|----------------|
| 瞬间大单 | 1-5秒 | ⚠️ 可能，但不完整 |
| 快速拉升 | 30秒-2分钟 | ❌ 只能看到片段 |
| 持续吸筹 | 10分钟-2小时 | ❌ 完全无法识别 |
| 震荡洗盘 | 30分钟-数小时 | ❌ 无法识别模式 |
| 出货 | 30分钟-数小时 | ❌ 只能看到片段 |

```python
def calculate_big_order_net(stock_data):
    """
    大单净量 = 大单买入金额 - 大单卖出金额
    
    大单定义：
    - 沪深主板：>100万
    - 创业板/科创板：>50万
    - 小盘股：>30万
    """
    big_buy = sum(amount for amount in orders if amount > 1_000_000 and is_buy)
    big_sell = sum(amount for amount in orders if amount > 1_000_000 and is_sell)
    return big_buy - big_sell

# 方法2：主力净流入法
def calculate_main_inflow(stock_data):
    """
    主力净流入 = 超大单 + 大单
    散户净流入 = 中单 + 小单
    """
    super_large = amount > 5_000_000   # 超大单
    large = 1_000_000 < amount <= 5_000_000  # 大单
    medium = 200_000 < amount <= 1_000_000    # 中单
    small = amount <= 200_000                 # 小单
    
    main_inflow = super_large_orders + large_orders
    retail_inflow = medium_orders + small_orders
    
    return main_inflow - retail_inflow

# 方法3：主动买卖法（需要逐笔数据）
def calculate_active_trade(stock_data):
    """
    主动买入：成交价 >= 卖一价（主动吃掉卖单）
    主动卖出：成交价 <= 买一价（主动吃掉买单）
    """
    active_buy = sum(amount for trade in trades if trade.price >= ask_price_1)
```

```python
def analyze_instant_window(current_data):
    """
    分析当前3秒窗口
    """
    delta = current_data['delta']
    
    # 因子1：成交额评分
    amount_score = calculate_amount_score(delta['amount'])
    
    # 因子2：成交量评分
    volume_score = calculate_volume_score(delta['volume'])
    
    # 因子3：价格变化评分
    price_score = calculate_price_change_score(delta['price_change_pct'])
    
    # 因子4：量价配合评分
    volume_price_score = calculate_volume_price_score(delta)
    
    # 综合评分
    instant_score = (
        amount_score * 0.35 +
        volume_score * 0.25 +
        price_score * 0.20 +
        volume_price_score * 0.20
    )
    
    # 判断是否为大单
    is_large_order = (
        delta['amount'] >= 300000 and  # 30万
        delta['volume'] >= 20000 and      # 200手
        instant_score >= 60
    )
    
    return {
        'score': instant_score,
        'is_large_order': is_large_order,
        'factors': {
            'amount_score': amount_score,
            'volume_score': volume_score,
            'price_score': price_score,
            'volume_price_score': volume_price_score
        }
    }
```

### 3.5 短期窗口分析（30秒）

```python
def analyze_short_window(stock_code, lookback=10):
    """
    分析最近30秒（10个3秒周期）
    """
    # 获取最近10个周期的数据
    recent_data = get_recent_periods(stock_code, lookback)
    
    if len(recent_data) < lookback:
        return {'score': 0, 'trend': 'unknown', 'pattern': 'insufficient_data'}
    
    # 计算累计指标
    cumulative = {
        'total_amount': sum(d['delta_amount'] for d in recent_data),
        'total_volume': sum(d['delta_volume'] for d in recent_data),
        'price_change': recent_data[-1]['price'] - recent_data[0]['price'],
        'price_change_pct': (recent_data[-1]['price'] - recent_data[0]['price']) / recent_data[0]['price'] * 100,
        'large_order_count': sum(1 for d in recent_data if d['is_large_order'])
    }
    
    # 趋势识别
    if cumulative['price_change_pct'] > 1.0 and cumulative['large_order_count'] >= 5:
        trend = 'rapid_rise'        # 快速拉升
        trend_score = 90
    elif cumulative['price_change_pct'] < -1.0 and cumulative['large_order_count'] >= 5:
        trend = 'rapid_fall'        # 快速下跌
        trend_score = 90
    elif abs(cumulative['price_change_pct']) < 0.3 and cumulative['large_order_count'] >= 3:
        trend = 'accumulation'      # 横盘吸筹
        trend_score = 70
    else:
        trend = 'normal'
        trend_score = 40
    
    # 量能模式
    avg_amount = cumulative['total_amount'] / lookback
    if avg_amount > 500000:  # 平均每3秒50万
        volume_pattern = 'high_volume'
        volume_score = 90
    elif avg_amount > 200000:  # 平均每3秒20万
        volume_pattern = 'medium_volume'
        volume_score = 60
    else:
        volume_pattern = 'low_volume'
        volume_score = 30
    
    # 综合评分
    short_term_score = trend_score * 0.6 + volume_score * 0.4
    
    return {
        'score': short_term_score,
        'trend': trend,
        'pattern': volume_pattern,
        'cumulative': cumulative
    }
```

### 3.6 中期窗口分析（5分钟）

```python
def analyze_medium_window(stock_code, lookback=100):
    """
    分析最近5分钟（100个3秒周期）
    """
    # 获取最近100个周期的数据
    recent_data = get_recent_periods(stock_code, lookback)
    
    if len(recent_data) < lookback * 0.8:  # 需要至少80%的数据
        return {'score': 0, 'behavior': 'unknown', 'confidence': 'low'}
    
    # 分段分析
    segments = {
        'first_2min': recent_data[0:40],      # 前2分钟
        'middle_2min': recent_data[40:80],     # 中间2分钟
        'last_1min': recent_data[80:100]       # 最后1分钟
    }
    
    # 计算各段特征
    segment_features = {}
    for name, segment in segments.items():
        segment_features[name] = {
            'total_amount': sum(d['delta_amount'] for d in segment),
            'total_volume': sum(d['delta_volume'] for d in segment),
            'price_change': segment[-1]['price'] - segment[0]['price'] if len(segment) > 1 else 0,
            'large_order_ratio': sum(1 for d in segment if d.get('is_large_order', False)) / len(segment)
        }
    
    # 行为模式识别
    behavior = recognize_behavior_pattern(segment_features)
    
    return {
        'score': behavior['score'],
        'behavior': behavior['type'],
        'confidence': behavior['confidence'],
        'segments': segment_features
    }

def recognize_behavior_pattern(segment_features):
    """
    识别主力行为模式
    """
    first = segment_features['first_2min']
    middle = segment_features['middle_2min']
    last = segment_features['last_1min']
    
    # 模式1：吸筹（前段放量，中段缩量，后段微涨）
    if (first['large_order_ratio'] > 0.3 and
        middle['large_order_ratio'] < 0.2 and
        last['price_change'] > 0):
        return {'type': 'accumulation', 'score': 80, 'confidence': 'medium'}
    
    # 模式2：拉升（后段快速放量上涨）
    if (last['price_change'] > 1.5 and
        last['large_order_ratio'] > 0.4 and
        last['total_amount'] > middle['total_amount'] * 2):
        return {'type': 'pump', 'score': 95, 'confidence': 'high'}
    
    # 模式3：出货（全程放量，价格下跌）
    if (first['large_order_ratio'] > 0.3 and
        last['price_change'] < -1.0 and
        last['large_order_ratio'] > 0.3):
        return {'type': 'dump', 'score': 90, 'confidence': 'high'}
    
    # 模式4：洗盘（震荡，量能不规则）
    if (abs(first['price_change']) < 0.5 and
        abs(middle['price_change']) < 0.5 and
        abs(last['price_change']) < 0.5 and
        first['large_order_ratio'] > 0.2):
        return {'type': 'wash', 'score': 60, 'confidence': 'low'}
    
    # 默认：散户交易
    return {'type': 'retail', 'score': 30, 'confidence': 'low'}
```
