# 分析中心系统设计方案 v2.1

> 版本：v2.1 | 日期：2026-04-13 | 状态：待评审
> 更新说明：领域分析与新闻分析业务相似，采用同构设计；各模块根据自身JSON结构独立设计

---

## 一、设计原则

### 1.1 核心原则

| 原则 | 说明 |
|-----|------|
| **独立设计** | 各分析模块根据自身JSON结构独立设计表结构和前端展示 |
| **不强制同构** | 不追求字段统一，保留各分析模块的业务特性 |
| **统一入口** | 四个分析模块共享一个分析中心入口，通过Tab切换 |
| **独立实现** | 各模块有自己的API、Service、前端组件 |
| **暂不关联** | 本期不实现跨模块数据关联功能 |

### 1.2 模块分类

| 分析模块 | 数据特点 | 设计策略 |
|---------|---------|---------|
| **新闻分析** | 消息集合（15-20条批量） | 参考现有设计，已稳定 |
| **领域分析** | 消息集合（30条批量） | 与新闻分析同构，字段扩展 |
| **涨停分析** | 单条（1只股票深度分析） | 独立设计，保留深度分析结构 |
| **公告分析** | 公告集合（多条批量） | 独立设计，侧重风险判定 |

### 2.3 数据表设计

```sql
CREATE TABLE `analysis_domain_detail_2026` (
  -- 主键与关联
  `id` BIGINT UNSIGNED NOT NULL AUTO_INCREMENT,
  `content_hash` VARCHAR(64) NOT NULL COMMENT '领域id：关键事件+时间的MD5',
  
  -- 领域信息
  `main_area` VARCHAR(64) NOT NULL COMMENT '主领域（科技/消费/医疗等）',
  `child_area` VARCHAR(64) NOT NULL COMMENT '子领域（AI/新能源等）',
  
  -- 原始事件信息（冗余存储）
  `event_time` DATETIME NOT NULL COMMENT '事件发生时间',
  `event_source` VARCHAR(100) DEFAULT '' COMMENT '事件来源（新华社/财联社等）',
  `key_event` VARCHAR(500) DEFAULT '' COMMENT '关键事件标题',
  `brief_desc` TEXT COMMENT '简要描述',
  
  -- AI评分字段（与新闻分析相同）
  `importance_score` TINYINT NOT NULL DEFAULT 0 COMMENT '重要程度评分（0-15）',
  `business_impact_score` SMALLINT NOT NULL DEFAULT 0 COMMENT '业务影响维度评分（-60至60）',
  `composite_score` SMALLINT NOT NULL DEFAULT 0 COMMENT '综合评分（重要程度×4+业务影响）',
  
  -- AI分类字段（与新闻分析相同）
  `news_size` ENUM('重大','大','中','小') NOT NULL DEFAULT '小' COMMENT '消息大小',
  `news_type` ENUM('利好','利空','中性') NOT NULL DEFAULT '中性' COMMENT '利空利好',
  
  -- 关联板块/概念/股票
  `sectors` JSON COMMENT '涉及板块（JSON数组）',
  `concepts` JSON COMMENT '涉及概念（JSON数组）',
  `stock_codes` JSON COMMENT '股票代码（JSON数组）',
  
  -- 原因分析（领域分析特有）
  `reason_analysis` TEXT COMMENT '原因分析',
  
  -- 深度分析（JSON数组）
  `deep_analysis` JSON COMMENT '深度分析（多维度）',
  
  -- 元数据
  `analysis_version` VARCHAR(32) DEFAULT '' COMMENT 'AI分析语料版本',
  `analysis_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'AI分析完成时间',
  `created_at` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '记录创建时间',
  
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_content_hash` (`content_hash`),
  KEY `idx_event_time` (`event_time`),
  KEY `idx_main_area` (`main_area`),
  KEY `idx_child_area` (`child_area`),
  KEY `idx_composite_score` (`composite_score`),
  KEY `idx_news_type` (`news_type`),
  KEY `idx_news_size` (`news_size`),
  -- 复合索引
  KEY `idx_area_time` (`main_area`, `child_area`, `event_time`),
  KEY `idx_time_type` (`event_time`, `news_type`),
  KEY `idx_time_score` (`event_time`, `composite_score`),
  -- 全文索引
  FULLTEXT KEY `ft_key_event_desc` (`key_event`, `brief_desc`, `reason_analysis`) WITH PARSER ngram
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci
  COMMENT='领域分析明细表（事件驱动分析）';
```

```
# 统计
domain:stats:{date}                  # Hash - 当日统计
domain:stats:{date}:{main_area}      # Hash - 某领域当日统计
```

```sql
  `stock_code` VARCHAR(20) NOT NULL COMMENT '股票代码',
  `stock_name` VARCHAR(64) DEFAULT '' COMMENT '股票简称',
  `notice_date` DATE NOT NULL COMMENT '公告日期',
  
  -- 公告基础信息
  `notice_title` VARCHAR(500) DEFAULT '' COMMENT '公告标题',
  `notice_content` TEXT COMMENT '公告内容',
  
  -- AI分析结果
  `risk_level` ENUM('高','中','低') DEFAULT '中' COMMENT '风险等级',
  `notice_type` ENUM('利好','利空','中性') DEFAULT '中性' COMMENT '消息类型',
  `judgment_basis` TEXT COMMENT '判定依据',
  `key_points` JSON COMMENT '关键要点 [字符串数组]',
  `short_term_impact` VARCHAR(500) DEFAULT '' COMMENT '短线影响',
  `medium_term_impact` VARCHAR(500) DEFAULT '' COMMENT '中线影响',
  
  -- 提取的索引字段
  `risk_score` TINYINT DEFAULT 50 COMMENT '风险评分(高=75,中=50,低=25)',
  `type_score` TINYINT DEFAULT 50 COMMENT '类型评分(利好=75,中性=50,利空=25)',
  
  -- 元数据
  `analysis_version` VARCHAR(32) DEFAULT '' COMMENT 'AI分析版本',
  `analysis_time` DATETIME DEFAULT CURRENT_TIMESTAMP,
  `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP,
  
  PRIMARY KEY (`id`),
  UNIQUE KEY `uk_content_hash` (`content_hash`),
  KEY `idx_notice_date` (`notice_date`),
  KEY `idx_stock_code` (`stock_code`),
  KEY `idx_stock_name` (`stock_name`),
  KEY `idx_risk_level` (`risk_level`),
  KEY `idx_notice_type` (`notice_type`),
  KEY `idx_risk_score` (`risk_score`),
  KEY `idx_time_type` (`notice_date`, `notice_type`),
  FULLTEXT KEY `ft_title_content` (`notice_title`, `notice_content`, `judgment_basis`) WITH PARSER ngram
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='公告分析明细表';
```

```python
# result_processor.py

class ResultProcessor:
    """统一分析结果处理器"""
    
    def process_news(self, json_data: str, source_table: str, version: str):
        """处理新闻分析结果（已有）"""
        pass
    
    def process_domain(self, json_data: str, main_area: str, child_area: str,
                       event_date: str, version: str = '1.0.0'):
        """处理领域分析结果（新增）
        
        Args:
            json_data: AI返回的JSON字符串
            main_area: 主领域
            child_area: 子领域
            event_date: 事件日期
            version: 分析版本
        """
        data = json.loads(json_data)
        messages = data.get('消息集合', [])
        
        for msg in messages:
            # 生成领域id：关键事件+时间的MD5
            key_event = msg.get('关键事件', '')
            event_time = msg.get('时间', '')
            content_hash = string_util.generate_md5(f"{key_event}_{event_time}")
            
            # 提取字段
            record = {
                'content_hash': content_hash,
                'main_area': main_area,
                'child_area': child_area,
                'event_time': msg.get('时间'),
                'event_source': msg.get('事件来源', ''),
                'key_event': msg.get('关键事件', ''),
                'brief_desc': msg.get('简要描述', ''),
                'news_type': msg.get('利空利好', '中性'),
                'news_size': msg.get('消息大小', '小'),
                'sectors': json.dumps(msg.get('涉及板块', '').split(','), ensure_ascii=False),
                'concepts': json.dumps(msg.get('涉及概念', '').split(','), ensure_ascii=False),
                'stock_codes': json.dumps(msg.get('股票代码', '').split(','), ensure_ascii=False),
                'reason_analysis': msg.get('原因分析', ''),
                'importance_score': int(msg.get('重要程度评分', 0)),
                'business_impact_score': int(msg.get('业务影响维度评分', 0)),
                'composite_score': int(msg.get('综合评分', 0)),
                'deep_analysis': json.dumps(msg.get('深度分析', []), ensure_ascii=False),
                'analysis_version': version,
            }
            
            # 写入MySQL
            self._save_domain_to_mysql(record)
            
            # 写入Redis
            self._save_domain_to_redis(record)
    
    def process_ztb(self, json_data: str, stock_name: str, trade_date: str,
                    stock_code: str = '', version: str = '1.0.0'):
        """处理涨停分析结果（新增）"""
        analysis = json.loads(json_data)
        
        # 生成content_hash
        content_hash = string_util.generate_md5(stock_name + trade_date)
        
        # 提取涨停时间判断时段
        zt_time = analysis.get('涨停时间', '')
        zt_time_range = self._get_zt_time_range(zt_time)
        
        # 提取索引字段
        sectors = [s.get('板块') for s in analysis.get('板块消息', [])]
        concepts = [c.get('概念') for c in analysis.get('概念消息', [])]
        leading_stocks = [l.get('龙头股') for l in analysis.get('龙头股消息', [])]
        expect_msgs = analysis.get('预期涨停消息', [])
        has_expect = 1 if expect_msgs else 0
        continuity = 1 if any(e.get('延续性') == '是' for e in expect_msgs) else 0
        
        record = {
            'content_hash': content_hash,
            'stock_name': stock_name,
            'stock_code': stock_code or analysis.get('股票代码', ''),
            'trade_date': trade_date,
            'zt_time': zt_time,
            'zt_time_range': zt_time_range,
            'stock_nature': analysis.get('股性分析', ''),
            'lhb_analysis': analysis.get('龙虎榜分析', ''),
            'sector_msg': json.dumps(analysis.get('板块消息', []), ensure_ascii=False),
            'concept_msg': json.dumps(analysis.get('概念消息', []), ensure_ascii=False),
            'leading_stock_msg': json.dumps(analysis.get('龙头股消息', []), ensure_ascii=False),
            'influence_msg': json.dumps(analysis.get('消息', []), ensure_ascii=False),
            'expect_msg': json.dumps(expect_msgs, ensure_ascii=False),
            'deep_analysis': json.dumps(analysis.get('深度分析', []), ensure_ascii=False),
            'sectors': json.dumps(sectors, ensure_ascii=False),
            'concepts': json.dumps(concepts, ensure_ascii=False),
            'leading_stocks': json.dumps(leading_stocks, ensure_ascii=False),
            'has_expect': has_expect,
            'continuity': continuity,
            'analysis_version': version,
        }
        
        self._save_ztb_to_mysql(record)
        self._save_ztb_to_redis(record)
    
    def process_notice(self, json_data: str, version: str = '1.0.0'):
        """处理公告分析结果（新增）"""
        data = json.loads(json_data)
        notices = data.get('公告集合', [])
        
        for notice in notices:
            content_hash = string_util.generate_md5(notice.get('公告id', ''))
            
            risk_map = {'高': 75, '中': 50, '低': 25}
            risk_level = notice.get('风险大小', '中')
            risk_score = risk_map.get(risk_level, 50)
            
            type_map = {'利好': 75, '中性': 50, '利空': 25}
            notice_type = notice.get('消息类型', '中性')
            type_score = type_map.get(notice_type, 50)
            
            record = {
                'content_hash': content_hash,
                'notice_id': notice.get('公告id', ''),
                'stock_code': notice.get('股票代码', ''),
                'stock_name': notice.get('股票名称', ''),
                'notice_date': notice.get('公告日期', datetime.now().strftime('%Y-%m-%d')),
                'notice_title': notice.get('公告标题', ''),
                'notice_content': notice.get('公告内容', ''),
                'risk_level': risk_level,
                'notice_type': notice_type,
                'judgment_basis': notice.get('判定依据', ''),
                'key_points': json.dumps(notice.get('关键要点', []), ensure_ascii=False),
                'short_term_impact': notice.get('短线影响', ''),
                'medium_term_impact': notice.get('中线影响', ''),
                'risk_score': risk_score,
                'type_score': type_score,
                'analysis_version': version,
            }
            
            self._save_notice_to_mysql(record)
            self._save_notice_to_redis(record)
    
    def _get_zt_time_range(self, zt_time: str) -> str:
        """根据涨停时间判断时段"""
        try:
            hour = int(zt_time.split(':')[0])
            minute = int(zt_time.split(':')[1])
            time_val = hour * 60 + minute
            
            # 09:30 = 570, 10:00 = 600, 14:00 = 840
            if time_val < 600:  # 09:30-10:00
                return 'early'
            elif time_val < 840:  # 10:00-14:00
                return 'mid'
            else:  # 14:00-15:00
                return 'late'
        except:
            return 'mid'
```
