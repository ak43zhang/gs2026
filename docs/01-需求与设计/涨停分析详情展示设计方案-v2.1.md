## 四、前端渲染逻辑

```javascript
// 板块消息渲染
function renderSectorMsg(sector_msg) {
    if (!sector_msg || sector_msg.length === 0) return '<p>无板块消息</p>';
    
    return sector_msg.map(s => `
        <div class="msg-group">
            <h4>🏭 ${s.板块}</h4>
            <ul>
                ${(s.板块相关消息 || []).map(m => `<li>${m}</li>`).join('')}
            </ul>
        </div>
    `).join('');
}

// 概念消息渲染
function renderConceptMsg(concept_msg) {
    if (!concept_msg || concept_msg.length === 0) return '<p>无概念消息</p>';
    
    return concept_msg.map(c => `
        <div class="msg-group">
            <h4>💡 ${c.概念}</h4>
            <ul>
                ${(c.概念相关消息 || []).map(m => `<li>${m}</li>`).join('')}
            </ul>
        </div>
    `).join('');
}

// 龙头股消息渲染
function renderLeadingStockMsg(leading_stock_msg) {
    if (!leading_stock_msg || leading_stock_msg.length === 0) return '<p>无龙头股消息</p>';
    
    return leading_stock_msg.map((l, idx) => `
        <div class="msg-group">
            <h4>👑 ${l.龙头股} ${idx === 0 ? '（领涨）' : ''}</h4>
            <ul>
                ${(l.龙头股相关消息 || []).map(m => `<li>${m}</li>`).join('')}
            </ul>
        </div>
    `).join('');
}

// 影响消息渲染
function renderInfluenceMsg(influence_msg) {
    if (!influence_msg || influence_msg.length === 0) return '<p>无影响消息</p>';
    
    // 按时间倒序
    const sorted = [...influence_msg].sort((a, b) => 
        new Date(b.影响消息时间) - new Date(a.影响消息时间)
    );
    
    return sorted.map(i => `
        <div class="influence-item">
            <div class="influence-date">📅 ${i.影响消息时间}</div>
            <div class="influence-content">${i.影响消息}</div>
        </div>
    `).join('');
}

// 预期消息渲染
function renderExpectMsg(expect_msg) {
    if (!expect_msg || expect_msg.length === 0) return '<p>无预期消息</p>';
    
    // 按时间倒序
    const sorted = [...expect_msg].sort((a, b) => 
        new Date(b.预期消息时间) - new Date(a.预期消息时间)
    );
    
    return sorted.map(e => `
        <div class="expect-item">
            <div class="expect-header">
                <span class="expect-date">📅 ${e.预期消息时间}</span>
                <span class="expect-continuity ${e.延续性 === '是' ? 'yes' : 'no'}">
                    ${e.延续性 === '是' ? '✅ 延续' : '❌ 不延续'}
                </span>
            </div>
            <div class="expect-content">${e.预期消息}</div>
        </div>
    `).join('');
}

// 深度分析渲染
function renderDeepAnalysis(deep_analysis) {
    if (!deep_analysis || deep_analysis.length === 0) return '<p>无深度分析</p>';
    
    const iconMap = {
        '技术面': '📈',
        '基本面': '📊',
        '资金面': '💰',
        '情绪面': '😊'
    };
    
    return deep_analysis.map(d => {
        const parts = d.split(':');
        const title = parts[0].trim();
        const content = parts.slice(1).join(':').trim();
        const icon = iconMap[title] || '•';
        
        // 将内容按逗号分割成多条
        const points = content.split(/[,，]/).filter(p => p.trim());
        
        return `
            <div class="analysis-block">
                <h4>${icon} ${title}</h4>
                <ul>
                    ${points.map(p => `<li>${p.trim()}</li>`).join('')}
                </ul>
            </div>
        `;
    }).join('');
}
```
