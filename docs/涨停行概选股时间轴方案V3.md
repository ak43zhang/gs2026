# 涨停行概选股时间轴方案设计 V3

## 一、需求理解（更新）

### 用户最新思路
1. **时间轴刷新机制**:
   - 点击时间轴刷新按钮 → 刷新时间点列表
   - **默认不自动刷新**（与数据监控一致）
   - 只有刷新页面或点击刷新按钮时才刷新时间轴

2. **时间轴数据来源**:
   - **先查Redis**: 获取Redis中存储的时间戳列表
   - **Redis无数据**: 从`monitor_gp_sssj_{date}`表获取
   - **与数据监控完全一致**

3. **涨停股票筛选**:
   - 选择时间点后，查询该时间点的`monitor_gp_sssj`数据
   - 筛选`is_zt=1`的股票
   - 进行行概分析

### 关键变化
- **时间轴数据**: 从Redis/MySQL获取实际时间戳（非固定间隔）
- **刷新机制**: 手动刷新，非实时自动刷新
- **数据一致性**: 与数据监控页面完全一致

## 三、方案设计 V3

### 3.1 核心流程

```
用户打开页面
    ↓
加载时间轴（手动/页面刷新）
    1. 调用 /api/ztb/timestamps
    2. 先查Redis: key="ztb:timestamps:{date}"
    3. Redis无: 查MySQL monitor_gp_sssj_{date} 获取DISTINCT time
    4. 时间戳列表写入Redis缓存
    ↓
时间轴显示可用时间点
    ↓
用户点击时间点（或拖动滑块）
    ↓
查询该时间点涨停股票
    1. 调用 /api/ztb/list-by-time?date={date}&time={time}
    2. 先查Redis: key="monitor_gp_sssj_{date}:{time}"
    3. Redis无: 查MySQL monitor_gp_sssj_{date} WHERE time={time}
    4. 筛选 is_zt=1 的股票
    ↓
行概分析
    ↓
展示结果
```

### 3.2 与数据监控对比

| 功能 | 数据监控 | 涨停选股（V3） |
|------|----------|----------------|
| **时间轴刷新** | 手动+自动 | **手动** ✅ |
| **时间戳来源** | Redis→MySQL | **Redis→MySQL** ✅ |
| **时间点数据** | Redis→MySQL | **Redis→MySQL** ✅ |
| **数据表** | monitor_gp_sssj | **monitor_gp_sssj** ✅ |
| **筛选条件** | 无 | **is_zt=1** ✅ |
| **后续分析** | 排行榜 | **行概分析** ✅ |

@ztb_bp.route('/api/ztb/timestamps')
def get_ztb_timestamps():
    """
    获取涨停选股时间轴时间戳列表
    
    逻辑:
    1. 先查Redis缓存
    2. Redis未命中，查MySQL monitor_gp_sssj_{date}表
    3. 返回DISTINCT time列表
    """
    try:
        date = request.args.get('date')
        if not date:
            return jsonify({'code': 400, 'message': '缺少date参数'}), 400
        
        date_str = date.replace('-', '')
        
        # 1. 先查Redis
        redis_key = f"ztb:timestamps:{date_str}"
        try:
            timestamps = redis_util.get(redis_key)
            if timestamps:
                logger.info(f"从Redis获取时间戳: {date}, 共{len(timestamps)}个")
                return jsonify({
                    'code': 0, 
                    'message': 'success',
                    'data': timestamps,
                    'source': 'redis'
                })
        except Exception as e:
            logger.warning(f"Redis查询失败: {e}")
        
        # 2. Redis未命中，查MySQL
        table_name = f"monitor_gp_sssj_{date_str}"
        sql = f"SELECT DISTINCT time FROM {table_name} ORDER BY time"
        
        try:
            df = pd.read_sql(sql, engine)
            timestamps = df['time'].tolist()
            logger.info(f"从MySQL获取时间戳: {date}, 共{len(timestamps)}个")
        except Exception as e:
            logger.error(f"MySQL查询失败: {e}")
            return jsonify({
                'code': 500, 
                'message': f'查询失败: {str(e)}'
            }), 500
        
        # 3. 写入Redis缓存（5分钟）
        try:
            redis_util.set(redis_key, timestamps, expire=300)
        except Exception as e:
            logger.warning(f"Redis写入失败: {e}")
        
        return jsonify({
            'code': 0,
            'message': 'success',
            'data': timestamps,
            'source': 'mysql'
        })
        
    except Exception as e:
        logger.error(f"获取时间戳失败: {e}")
        return jsonify({'code': 500, 'message': str(e)}), 500

### 4.2 服务层实现

```python
# ztb_analysis_service.py

def get_ztb_list_by_time_v3(date: str, time_str: str) -> Dict:
    """
    V3: 获取指定时间点的涨停股票列表
    
    与数据监控完全一致的数据查询逻辑:
    1. 先查Redis
    2. Redis未命中查MySQL
    3. 筛选is_zt=1
    
    Args:
        date: 日期，如 "2026-04-29"
        time_str: 时间，如 "10:00:00"
    
    Returns:
        {
            'zt_stocks': [...],      # 涨停股票列表
            'industries': [...],      # 行业分布
            'concepts': [...],        # 概念分布
            'stats': {...}            # 统计信息
        }
    """
    date_str = date.replace('-', '')
    redis_key = f"monitor_gp_sssj_{date_str}:{time_str}"
    
    df = None
    source = None
    
    # 1. 先查Redis（与数据监控完全一致）
    try:
        df = redis_util.load_dataframe_by_key(redis_key)
        if df is not None and not df.empty:
            source = 'redis'
            logger.info(f"从Redis获取数据: {redis_key}, {len(df)}条")
    except Exception as e:
        logger.warning(f"Redis查询失败: {e}")
    
    # 2. Redis未命中，查MySQL（与数据监控完全一致）
    if df is None or df.empty:
        try:
            table_name = f"monitor_gp_sssj_{date_str}"
            sql = f"""
                SELECT * FROM {table_name}
                WHERE time = %s
            """
            df = pd.read_sql(sql, engine, params=(time_str,))
            source = 'mysql'
            logger.info(f"从MySQL获取数据: {table_name}, time={time_str}, {len(df)}条")
        except Exception as e:
            logger.error(f"MySQL查询失败: {e}")
            raise
    
    # 3. 数据为空
    if df is None or df.empty:
        return {
            'zt_stocks': [],
            'industries': [],
            'concepts': [],
            'stats': {
                'total': 0,
                'source': None,
                'time': time_str,
                'date': date
            }
        }
    
    # 4. 筛选is_zt=1的股票
    if 'is_zt' not in df.columns:
        logger.error(f"数据缺少is_zt字段，可用字段: {df.columns.tolist()}")
        raise ValueError("数据缺少is_zt字段")
    
    zt_df = df[df['is_zt'] == 1].copy()
    total_zt = len(zt_df)
    
    logger.info(f"时间点{time_str}涨停股票: {total_zt}/{len(df)}")
    
    # 5. 处理股票列表
    zt_stocks = []
    for _, row in zt_df.iterrows():
        zt_stocks.append({
            'stock_code': str(row.get('stock_code', '')).zfill(6),
            'stock_name': row.get('short_name', ''),
            'price': float(row.get('price', 0)),
            'change_pct': float(row.get('change_pct', 0)),
            'main_net_amount': float(row.get('main_net_amount', 0)),
            'cumulative_main_net': float(row.get('cumulative_main_net', 0)),
            'time': time_str
        })
    
    # 6. 行概分析
    industries = analyze_industries_v3(zt_stocks)
    concepts = analyze_concepts_v3(zt_stocks)
    
    return {
        'zt_stocks': zt_stocks,
        'industries': industries,
        'concepts': concepts,
        'stats': {
            'total': total_zt,
            'source': source,
            'time': time_str,
            'date': date,
            'all_stocks': len(df)
        }
    }


def analyze_industries_v3(stocks: List[Dict]) -> List[Dict]:
    """分析行业分布V3"""
    if not stocks:
        return []
    
    # 获取行业信息
    stock_codes = [s['stock_code'] for s in stocks]
    industry_map = get_stock_industry_map(stock_codes)  # 从缓存或数据库
    
    # 统计
    industry_stats = {}
    for stock in stocks:
        code = stock['stock_code']
        industry = industry_map.get(code, '未知')
        
        if industry not in industry_stats:
            industry_stats[industry] = {
                'name': industry,
                'count': 0,
                'stocks': []
            }
        
        industry_stats[industry]['count'] += 1
        industry_stats[industry]['stocks'].append({
            'code': code,
            'name': stock['stock_name']
        })
    
    # 排序并返回
    result = sorted(
        industry_stats.values(),
        key=lambda x: x['count'],
        reverse=True
    )
    
    return result


def analyze_concepts_v3(stocks: List[Dict]) -> List[Dict]:
    """分析概念分布V3"""
    if not stocks:
        return []
    
    # 获取概念信息
    stock_codes = [s['stock_code'] for s in stocks]
    concept_map = get_stock_concept_map(stock_codes)  # 从缓存或数据库
    
    # 统计
    concept_stats = {}
    for stock in stocks:
        code = stock['stock_code']
        concepts = concept_map.get(code, [])
        
        for concept in concepts:
            if concept not in concept_stats:
                concept_stats[concept] = {
                    'name': concept,
                    'count': 0,
                    'stocks': []
                }
            
            concept_stats[concept]['count'] += 1
            concept_stats[concept]['stocks'].append({
                'code': code,
                'name': stock['stock_name']
            })
    
    # 排序并返回
    result = sorted(
        concept_stats.values(),
        key=lambda x: x['count'],
        reverse=True
    )
    
    return result[:20]  # 只返回前20个概念
```
