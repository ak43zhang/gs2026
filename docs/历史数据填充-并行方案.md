# 历史数据填充 - 并行方案

## 目标
- 填充2026-04-29全部5000只股票的主力净额和累计主力净额
- 时间要求：30分钟内完成
- 方案：5个并行进程，每进程1000只股票

---

## 方案设计

### 并行策略

```
总股票: 5000只
进程数: 5个
每进程: 1000只
预计时间: 30分钟 (每只股票约0.36秒)
```

### 单进程处理逻辑

```python
# 每只股票处理流程
for stock_code in batch_1000:
    # 1. 查询该股票所有数据 (约500-1000条)
    SELECT time, amount FROM table WHERE stock_code = 'xxx' ORDER BY time
    
    # 2. Python计算 (内存中)
    for each row:
        main_net = (current_amount - prev_amount) * 0.3
        cumulative += main_net
    
    # 3. 批量更新 (一次性UPDATE)
    UPDATE table SET main_net_amount = ..., cumulative_main_net = ... 
    WHERE stock_code = 'xxx' AND time IN (...)
```

### 关键优化

1. **批量查询** - 每只股票一次性查询所有数据
2. **内存计算** - Python中计算，避免SQL窗口函数
3. **批量更新** - 每只股票一次性UPDATE所有时间点
4. **进程隔离** - 5个进程处理不同股票，无锁竞争

## 实施步骤

### 步骤1：创建主控脚本

**文件**: `fill_master.py`

```python
#!/usr/bin/env python3
"""
主控脚本 - 分发任务给5个工作进程
"""
import subprocess
import sys
import time

print("=" * 60)
print("并行填充主控 - 5000只股票分5进程")
print("=" * 60)

start_time = time.time()

# 启动5个工作进程
processes = []
for i in range(5):
    worker_id = i + 1
    print(f"\n启动工作进程 {worker_id}/5...")
    
    p = subprocess.Popen([
        sys.executable, 'fill_worker.py', 
        str(worker_id),  # 进程ID
        str(5)            # 总进程数
    ])
    processes.append(p)

print(f"\n所有工作进程已启动，等待完成...")

# 等待所有进程完成
for i, p in enumerate(processes, 1):
    p.wait()
    print(f"  进程 {i}/5 完成")

elapsed = time.time() - start_time
print(f"\n{'='*60}")
print(f"全部完成! 总用时: {elapsed/60:.1f} 分钟")
```

### 步骤2：创建工作进程

**文件**: `fill_worker.py`

```python
#!/usr/bin/env python3
"""
工作进程 - 处理1000只股票
"""
import pymysql
import sys
import time

worker_id = int(sys.argv[1])
total_workers = int(sys.argv[2])

print(f"[进程{worker_id}] 启动...")

conn = pymysql.connect(
    host='192.168.0.101', port=3306, user='root', password='123456',
    database='gs', charset='utf8mb4'
)

cursor = conn.cursor()

# 获取该进程负责的股票 (按stock_code哈希分配)
cursor.execute(f"""
    SELECT DISTINCT stock_code 
    FROM monitor_gp_sssj_20260429 
    WHERE amount > 0
    AND MOD(CRC32(stock_code), {total_workers}) = {worker_id - 1}
""")

stocks = [r[0] for r in cursor.fetchall()]
print(f"[进程{worker_id}] 负责 {len(stocks)} 只股票")

total_updated = 0
start = time.time()

for i, code in enumerate(stocks, 1):
    try:
        # 1. 查询该股票所有数据
        cursor.execute(f"""
            SELECT time, amount 
            FROM monitor_gp_sssj_20260429 
            WHERE stock_code = %s 
            ORDER BY time
        """, (code,))
        
        rows = cursor.fetchall()
        if len(rows) < 2:
            continue
        
        # 2. 内存计算
        prev_amount = None
        cumulative = 0
        updates = []
        
        for time_str, amount in rows:
            try:
                amount = float(amount) if amount else 0
            except:
                amount = 0
            
            if prev_amount is None:
                main_net = 0
            else:
                main_net = (amount - prev_amount) * 0.3
            
            cumulative += main_net
            prev_amount = amount
            updates.append((main_net, cumulative, code, time_str))
        
        # 3. 批量更新 - 使用executemany
        cursor.executemany("""
            UPDATE monitor_gp_sssj_20260429 
            SET main_net_amount = %s,
                cumulative_main_net = %s
            WHERE stock_code = %s AND time = %s
        """, updates)
        
        conn.commit()
        total_updated += len(updates)
        
        # 每100只报告进度
        if i % 100 == 0:
            elapsed = time.time() - start
            speed = i / elapsed
            remaining = (len(stocks) - i) / speed if speed > 0 else 0
            print(f"[进程{worker_id}] {i}/{len(stocks)} 只，"
                  f"已更新 {total_updated} 条，"
                  f"用时 {elapsed/60:.1f}分钟，"
                  f"预计剩余 {remaining/60:.1f}分钟")
                  
    except Exception as e:
        print(f"[进程{worker_id}] {code} 失败: {e}")

conn.close()

elapsed = time.time() - start
print(f"[进程{worker_id}] 完成! 更新 {total_updated} 条，用时 {elapsed/60:.1f}分钟")
```

### 步骤3：创建进度监控脚本

**文件**: `monitor_progress.py`

```python
#!/usr/bin/env python3
"""
实时监控填充进度
"""
import pymysql
import time
import sys

print("=" * 60)
print("填充进度监控")
print("=" * 60)

conn = pymysql.connect(
    host='192.168.0.101', port=3306, user='root', password='123456',
    database='gs', charset='utf8mb4'
)

cursor = conn.cursor()

# 获取总股票数
cursor.execute("""
    SELECT COUNT(DISTINCT stock_code) 
    FROM monitor_gp_sssj_20260429 
    WHERE amount > 0
""")
total_stocks = cursor.fetchone()[0]

print(f"\n总股票数: {total_stocks}")
print(f"目标: 30分钟内完成")
print(f"\n{'='*60}")

# 实时监控
prev_main = 0
prev_cum = 0
start_time = time.time()

while True:
    cursor.execute("""
        SELECT 
            COUNT(*),
            SUM(CASE WHEN main_net_amount != 0 THEN 1 ELSE 0 END),
            SUM(CASE WHEN cumulative_main_net != 0 THEN 1 ELSE 0 END)
        FROM monitor_gp_sssj_20260429
    """)
    
    row = cursor.fetchone()
    total = row[0]
    main = row[1]
    cum = row[2]
    
    elapsed = time.time() - start_time
    
    # 计算速度
    main_speed = (main - prev_main) / 60  # 条/秒
    cum_speed = (cum - prev_cum) / 60
    
    # 预估剩余时间
    if main > 0:
        eta_seconds = (total - main) / main_speed if main_speed > 0 else 999999
        eta_minutes = eta_seconds / 60
    else:
        eta_minutes = 999
    
    # 清屏并打印
    sys.stdout.write('\033[2J\033[H')  # 清屏
    print(f"{'='*60}")
    print(f"填充进度监控 - 运行时间: {elapsed/60:.1f}分钟")
    print(f"{'='*60}")
    print(f"\n总记录: {total:,}")
    print(f"已填充 main_net_amount: {main:,} ({main/total*100:.2f}%)")
    print(f"已填充 cumulative_main_net: {cum:,} ({cum/total*100:.2f}%)")
    print(f"\n速度: {main_speed:.0f} 条/秒")
    print(f"预估剩余时间: {eta_minutes:.1f} 分钟")
    
    if main >= total * 0.99:
        print(f"\n{'='*60}")
        print("填充完成!")
        break
    
    prev_main = main
    prev_cum = cum
    
    time.sleep(10)  # 每10秒更新

conn.close()
```
