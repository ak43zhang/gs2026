## 修复方案

### 方案1：智能日期选择（推荐）

**思路**：默认选择最近有数据的日期

```javascript
// 修改stock_picker.html中的initZtbPanel函数

async function initZtbPanel() {
    if (ztbAllIndustries.length === 0) {
        // 【修改】智能选择最近有数据的日期
        const today = new Date();
        const todayStr = today.toISOString().split('T')[0];
        
        // 先尝试今天
        let dateToUse = todayStr;
        
        // 查询今天是否有数据
        const hasDataToday = await checkDateHasData(todayStr);
        if (!hasDataToday) {
            // 今天无数据，使用昨天
            const yesterday = new Date(today);
            yesterday.setDate(yesterday.getDate() - 1);
            dateToUse = yesterday.toISOString().split('T')[0];
        }
        
        document.getElementById('ztbDate').value = dateToUse;
        loadZtbTags();
    }
}

// 新增函数：检查日期是否有数据
async function checkDateHasData(dateStr) {
    try {
        const response = await fetch(`/api/stock-picker/ztb-tags?date=${dateStr.replace(/-/g, '')}`);
        const data = await response.json();
        return data.code === 0 && data.data && data.data.total_zt > 0;
    } catch (e) {
        return false;
    }
}
```

### 方案3：前端提示+手动选择

**思路**：今天无数据时，提示用户选择其他日期

```javascript
// 修改loadZtbTags函数

async function loadZtbTags() {
    const dateInput = document.getElementById('ztbDate');
    const date = dateInput.value.replace(/-/g, '');
    
    if (!date) {
        alert('请选择日期');
        return;
    }
    
    document.getElementById('ztbEmptyState').innerHTML = '<div class="loading">加载中...</div>';
    
    try {
        const response = await fetch(`/api/stock-picker/ztb-tags?date=${date}`);
        const data = await response.json();
        
        if (data.code !== 0) {
            throw new Error(data.message || '加载失败');
        }
        
        const result = data.data;
        
        // 【新增】无数据时提示
        if (result.total_zt === 0) {
            document.getElementById('ztbEmptyState').innerHTML = `
                <div class="empty-state">
                    <div class="empty-icon">📅</div>
                    <div class="empty-text">${date}无涨停数据</div>
                    <div class="empty-subtext">请选择其他日期</div>
                    <button onclick="selectPrevDay()" class="btn-primary">查看前一天</button>
                </div>
            `;
            return;
        }
        
        // ... 原有代码 ...
    } catch (error) {
        // ... 原有代码 ...
    }
}

// 新增函数：选择前一天
function selectPrevDay() {
    const dateInput = document.getElementById('ztbDate');
    const currentDate = new Date(dateInput.value);
    currentDate.setDate(currentDate.getDate() - 1);
```

## 推荐方案

| 方案 | 优点 | 缺点 | 推荐度 |
|------|------|------|--------|
| 方案1：智能日期 | 用户体验好，自动选择有数据日期 | 需要额外请求 | ⭐⭐⭐⭐⭐ |
| 方案2：后端回退 | 逻辑集中，前端无改动 | 用户可能不知道日期被修改 | ⭐⭐⭐⭐ |
| 方案3：提示+手动 | 明确告知用户 | 需要用户操作 | ⭐⭐⭐ |

**最终推荐**：方案1 + 方案3 结合
- 默认自动选择最近有数据的日期
- 如果仍无数据，显示友好提示
