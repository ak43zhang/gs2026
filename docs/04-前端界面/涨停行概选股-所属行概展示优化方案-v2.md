### JavaScript交互

```javascript
// 标签展开/收起状态
let tagsExpanded = false;

function toggleTagsExpand() {
    const grid = document.getElementById('tagsGrid');
    const btn = document.getElementById('expandBtn');
    const totalTags = grid.querySelectorAll('.tag-item').length;
    
    tagsExpanded = !tagsExpanded;
    
    if (tagsExpanded) {
        // 展开 - 显示全部
        grid.classList.remove('collapsed');
        btn.textContent = '收起';
    } else {
        // 收起 - 只显示第一行（5个）
        grid.classList.add('collapsed');
        const hiddenCount = totalTags - 5;
        btn.textContent = `+${hiddenCount}`;
    }
}

// 初始化标签展示
function initTagsDisplay() {
    const grid = document.getElementById('tagsGrid');
    const btn = document.getElementById('expandBtn');
    const totalTags = grid.querySelectorAll('.tag-item').length;
    
    // 如果标签数 <= 5，不需要展开按钮
    if (totalTags <= 5) {
        btn.style.display = 'none';
        grid.classList.remove('collapsed');
    } else {
        // 显示 "+N"
        btn.textContent = `+${totalTags - 5}`;
    }
}

// 渲染标签时调用
function renderTags(industries, concepts) {
    const grid = document.getElementById('tagsGrid');
    
    // 清空
    grid.innerHTML = '';
    
    // 添加行业标签（优先）
    industries.forEach(ind => {
        const tag = document.createElement('span');
        tag.className = 'tag-item tag-industry';
        tag.textContent = `🏭${ind}`;
        grid.appendChild(tag);
    });
    
    // 添加概念标签
    concepts.forEach(con => {
        const tag = document.createElement('span');
        tag.className = 'tag-item tag-concept';
        tag.textContent = `💡${con}`;
        grid.appendChild(tag);
    });
    
    // 初始化展示
    initTagsDisplay();
}
```
